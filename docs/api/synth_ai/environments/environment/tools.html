 44class ToolResult(BaseModel):
 45    """
 46    Represents the result of executing an environment tool.
 47    
 48    This class standardizes the response format for all tool executions,
 49    providing a consistent interface for success/failure status, return
 50    values, and error information. This makes it easy for agents and
 51    environment systems to handle tool execution results uniformly.
 52    
 53    Attributes:
 54        ok: Whether the tool execution was successful
 55        payload: The return value from the tool (None if failed or no return value)
 56        error: Error message if execution failed (None if successful)
 57    
 58    Example:
 59        >>> # Successful tool execution
 60        >>> success_result = ToolResult(
 61        ...     ok=True,
 62        ...     payload={"new_position": [5, 3], "items_collected": ["key"]},
 63        ...     error=None
 64        ... )
 65        
 66        >>> # Failed tool execution
 67        >>> error_result = ToolResult(
 68        ...     ok=False,
 69        ...     payload=None,
 70        ...     error="Cannot move north: wall blocking path"
 71        ... )
 72        
 73        >>> # Success with no return value
 74        >>> simple_success = ToolResult(ok=True)
 75    """
 76    ok: bool
 77    payload: Any | None = None
 78    error: str | None = None

 81class AbstractTool(ABC):
 82    """
 83    Abstract base class for all environment tools.
 84    
 85    Tools are the primary mechanism for agents to interact with environments.
 86    Each tool represents a specific action or capability that an agent can
 87    invoke, such as moving, picking up items, or examining objects.
 88    
 89    Tools define their own call and result schemas using Pydantic models,
 90    enabling automatic validation and documentation generation. This ensures
 91    that agents provide valid inputs and receive structured outputs.
 92    
 93    The tool system supports:
 94    - Type-safe argument validation
 95    - Automatic error handling and reporting  
 96    - Consistent result formatting
 97    - Dynamic tool registration and discovery
 98    
 99    Attributes:
100        name: Unique identifier for this tool (used in EnvToolCall.tool)
101        call_schema: Pydantic model defining valid arguments for this tool
102        result_schema: Pydantic model defining the structure of results
103    
104    Example:
105        >>> class MoveTool(AbstractTool):
106        ...     name = "move"
107        ...     call_schema = MoveArgs  # Pydantic model with 'direction' field
108        ...     
109        ...     async def __call__(self, call: EnvToolCall) -> ToolResult:
110        ...         direction = call.args['direction']
111        ...         # Perform movement logic
112        ...         if valid_move:
113        ...             return ToolResult(ok=True, payload=new_position)
114        ...         else:
115        ...             return ToolResult(ok=False, error="Invalid move")
116        
117        >>> # Register and use the tool
118        >>> move_tool = MoveTool()
119        >>> register_tool(move_tool)
120        >>> call = EnvToolCall(tool="move", args={"direction": "north"})
121        >>> result = await move_tool(call)
122    """
123    name: str
124    call_schema: Type[BaseModel]
125    result_schema: Type[BaseModel] = ToolResult
126
127    @abstractmethod
128    async def __call__(self, call: EnvToolCall) -> ToolResult:
129        """
130        Execute the tool with the given tool call.
131        
132        This method contains the core logic for the tool's functionality.
133        It should:
134        1. Validate the tool call arguments (using call_schema)
135        2. Perform the requested action
136        3. Return a ToolResult with appropriate success/failure status
137        
138        Args:
139            call: The tool call containing the tool name and arguments
140            
141        Returns:
142            ToolResult: Result of tool execution with success status,
143                payload data, and any error messages
144                
145        Raises:
146            ValidationError: If call arguments don't match call_schema
147            EnvironmentError: If tool execution fails due to environment state
148            
149        Example:
150            >>> call = EnvToolCall(tool="move", args={"direction": "east"})
151            >>> result = await tool(call)
152            >>> if result.ok:
153            ...     print(f"Moved to: {result.payload['position']}")
154            ... else:
155            ...     print(f"Move failed: {result.error}")
156        """
157        ...

163def register_tool(tool: AbstractTool) -> None:
164    """
165    Register a tool instance for use in environments.
166    
167    This function adds a tool to the global registry, making it available
168    for environments to use when processing agent tool calls. Tools must
169    be registered before they can be invoked by agents.
170    
171    The registry uses the tool's name attribute as the key, so tool names
172    must be unique across all registered tools.
173    
174    Args:
175        tool: The tool instance to register. Must have a unique name.
176        
177    Raises:
178        ValueError: If a tool with the same name is already registered
179        TypeError: If tool is not an instance of AbstractTool
180        
181    Example:
182        >>> class LookTool(AbstractTool):
183        ...     name = "look"
184        ...     # ... implementation
185        
186        >>> look_tool = LookTool()
187        >>> register_tool(look_tool)
188        >>> 
189        >>> # Now agents can use: EnvToolCall(tool="look", args={})
190        
191    Note:
192        Tools are typically registered during environment initialization
193        or module import. Once registered, tools remain available for
194        the duration of the application session.
195    """
196    TOOL_REGISTRY[tool.name] = tool

Represents an agent-requested call to an environment tool.

This class encapsulates an action that an AI agent wants to perform in an environment. Tool calls consist of a tool name and a dictionary of arguments that will be passed to the tool for execution.

The tool call system provides a standardized way for agents to interact with environments, making it easy to:

  • Validate agent actions before execution
  • Log and trace agent behavior
  • Implement complex multi-step actions
  • Handle errors and provide feedback

Attributes: tool: The name of the tool to invoke (must be registered in environment) args: Arguments to pass to the tool, with argument names as keys

Example:

Simple movement action

move_call = EnvToolCall(tool="move", args={"direction": "north"})

>>> # Complex action with multiple parameters
>>> craft_call = EnvToolCall(
...     tool="craft_item",
...     args={"item": "sword", "materials": ["iron", "wood"], "quantity": 1}
... )

>>> # Action with no arguments
>>> look_call = EnvToolCall(tool="look", args={})
